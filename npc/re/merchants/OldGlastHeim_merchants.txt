OnInit:
// === Reset Feature ===
// 0 = Disabled (Official)
// 1 = Normal Reset
// 2 = Restricted Reset
.enchant_reset = 0;
if(.enchant_reset) {
	// === Type of Boots resetable ===
	// 0 = All (default)
	// 1 = Unslotted only
	// 2 = Slotted only
	.reset_boots = 0;
	
	// === Card Slot behaviour on reset ===
	// 0 = Card will not be removed (default)
	// 1 = Card will be removed
	.reset_card = 0;
	
	// === Restricted Reset ===
	if(.enchant_reset == 2) {
		// 0 = Only 1 Enchantment
		// 1 = Required Refine
		.reset_restrict = 0;
		
		if(.reset_restrict)
			// === Required Refine for Reset === 
			.reset_refine = 7;
	}
	// === Chance of Reset == 
	// > Define here the chance for the reset
	// 50% = Default
	.reset_chance = 50;

	// === Reset Price ===
	// 0 = Disabled
	// 1 = Enabled
	.reset_price = 1;
	if(.reset_price)
		// === Reset Price List === 
		// Note: You can put as many as you want
		setarray .reset_cost[0],
		// 			Format: <Item ID>,<Cost>
								6607,5,	 // Temporal Crystal
								6755,10, // Coagulated Spell 
								6608,10; // Contaminated Magic
}
end;
}
